package P3.Monticulos;

public class EDBinaryHeap <T extends Comparable<T>> implements EDPriorityQueue<T>{
	
	protected T [] elementos;
	protected int numElementos;
	
	
	public EDBinaryHeap(int numMaxElementos) {
		this.elementos = (T[]) new Comparable[numMaxElementos];
		this.numElementos = 0;
	}

	@Override
	public boolean add(T info) {
		if(info == null) {
			throw new NullPointerException();
		}
		if(elementos.length <= numElementos) {
			return false;
		}
		if(isEmpty()) {
			elementos[0] = info;
		} else {
			elementos[numElementos] = info;
			filtradoAscendente(numElementos);
		}
		numElementos++;
		return true;
	}

	private void filtradoAscendente(int posHijo) {
		int posPadre = (posHijo-1)/2;
		if(elementos[posPadre].compareTo(elementos[posHijo]) > 0) {
			T aux = elementos[posHijo];
			elementos[posHijo] = elementos[posPadre];
			elementos[posPadre] = aux;
			filtradoAscendente(posPadre);
		}
	}

	@Override
	public T poll() {
		if(isEmpty()) {
			return null;
		}
		T sacar = elementos[0];
		elementos[0] = elementos[numElementos];
		elementos[numElementos] = null;
		filtradoDescendente(0);
		return sacar;
	}

	private void filtradoDescendente(int posPadre) {
		if(elementos[posPadre] == null) {
			posPadre = 0;
		}
		int menorHijos = getMinHijo(posPadre);
		while(elementos[posPadre].compareTo(elementos[menorHijos]) < 0) {
			T aux = elementos[posPadre];
			elementos[posPadre] = elementos[menorHijos];
			elementos[menorHijos] = aux;
			filtradoDescendente(menorHijos);
		}
	}

	private int getMinHijo(int posPadre) {
		int posHijoIzq = 2*posPadre + 1;
		int posHijoDer = 2* posPadre + 2;
		if (posHijoIzq >= numElementos){
			return -1;
		} else if(posHijoDer >= numElementos && posHijoIzq < numElementos){
			return posHijoIzq;
		}
		if(elementos[posHijoIzq].compareTo(elementos[posHijoDer]) <= 0) {
			return posHijoIzq;
		} else {
			return posHijoDer;
		}
	}

	@Override
	public boolean remove(T info) {
		int posicion = searchElement(info);
		if(posicion != 0) {
			elementos[posicion] = elementos [numElementos];
			elementos[numElementos] = null;
			numElementos--;
			filtradoDescendente(posicion);
			return true;
		}
		return false;
	}

	private int searchElement(T info) {
		if(info == null) {
			throw new NullPointerException();
		}
		for (int i = 0; i < numElementos; i++) {
			if(info.equals(elementos[i])) {
				return i;
			}
		}
		return 0;
	}

	@Override
	public boolean isEmpty() {
		if(numElementos != 0) {
			return true;
		}
		return false;
	}

	@Override
	public void clear() {
		numElementos = 0;
		
	}

	@Override
	public boolean cambiarPrioridad(int pos, T elemento) {
		// TODO Auto-generated method stub
		return false;
	}

}
